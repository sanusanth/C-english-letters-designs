
public class sanus   
{   
public static void main(String args[])   
{   
//i for rows and j for columns      
//row denotes the number of rows you want to print  
int i, j;   
//outer loop for rows  
for(i=0; i<7; i++)   
{   
//inner loop for columns  
for(j=0; j<i; j++)   
{   
//prints stars   
System.out.print("* ");   
}   

System.out.println();   
}
